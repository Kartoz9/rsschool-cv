```function toWeirdCase(string){
    const words = string.split(' '); 

  const resultWords = words.map((word) => {
    return word
      .split('')
      .map((char, index) => {
        if (index % 2 === 0) {
          return char.toUpperCase(); 
        } else {
          return char.toLowerCase(); 
        }
      })
      .join(''); 
  });

  return resultWords.join(' '); 
}
```
